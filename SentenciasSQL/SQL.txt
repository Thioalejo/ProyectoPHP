Creacion de tabla tendero

CREATE TABLE `carnesfrias`.`Tendero` ( `Nit` INT NOT NULL AUTO_INCREMENT , `Nombre_Completo` VARCHAR(200) NOT NULL , `Ciudad` VARCHAR(50) NOT NULL , `Barrio` VARCHAR(50) NOT NULL , `Direccion` VARCHAR(50) NOT NULL , `Telefono` BIGINT NOT NULL , PRIMARY KEY (`Nit`)) ENGINE = InnoDB;
